\begin{document}
    \input{operation.tex}


    \begin{tikzpicture}[
            >=stealth,
            scale=\myscale,
        ]

        % Output grid --------------------------------------------------------------
        \begin{pgfonlayer}{background}
            \begin{scope}[
                    yshift=(0 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[output sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                \draw[output cell] (2, 2) rectangle +(1, 1);

                \coordinate (output cell) at (2.5, 2.5);

            \end{scope}
        \end{pgfonlayer}

        \begin{pgfonlayer}{foreground}

            % Optional additional input grid(s) ------------------------------------
            \foreach \l in {0, 1, 2} {
                \begin{scope}[
                        yshift=((2 + (\l * 0.2)) * \mysheetspacing),
                        yslant=\myyslant,
                        xslant=\myxslant,
                    ]

                    \fill[input sheet, opacity=0.2] (-0.5, -0.5) rectangle (5.5, 5.5);
                    \draw[grid, opacity=0.4] (-0.25, -0.25) grid (5.25, 5.25);

                    \draw[input cell, opacity=0.4] (2, 2) rectangle +(1, 1);

                \end{scope}
            }


            % Input grid -----------------------------------------------------------
            \begin{scope}[
                    yshift=(3 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[input sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                \draw[input cell] (2, 2) rectangle +(1, 1);

                \coordinate (input cell) at (2.5, 2.5);

            \end{scope}
        \end{pgfonlayer}

        \begin{pgfonlayer}{main}
            \draw[arrow, shorten >= \myscale * 1.0cm]
                (input cell) -- (output cell);
        \end{pgfonlayer}

    \end{tikzpicture}
