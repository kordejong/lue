    \begin{tikzpicture}[

            scale=\myscale,

        \def\mypoints{(0, 4), (1, 3), (1, 4), (2, 2), (2, 3), (2, 4), (3, 4)}

        % Output grid --------------------------------------------------------------
        \begin{pgfonlayer}{background}
            \begin{scope}[
                    yshift=(0 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[output sheet] (-0.5, -0.5) rectangle (5.5, 5.5);

                \draw[output cell] (2, 2) rectangle +(1, 1);

                \coordinate (output cell) at (2.5, 2.5);

            \end{scope}
        \end{pgfonlayer}

        % Input values grid --------------------------------------------------------
        \begin{pgfonlayer}{main}
            \begin{scope}[
                    yshift=(2 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[input sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                % (0, 4), (1, 3), (1, 4), (2, 2), (2, 3), (2, 4), (3, 4)

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (0, 4) rectangle +(1, 1);
                \coordinate ({input cell 1}) at (0.5, 4.5);

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (1, 3) rectangle +(1, 1);
                \coordinate ({input cell 2}) at (1.5, 3.5);

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (1, 4) rectangle +(1, 1);
                \coordinate ({input cell 3}) at (1.5, 4.5);

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (2, 2) rectangle +(1, 1);
                \coordinate ({input cell 4}) at (2.5, 2.5);

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (2, 3) rectangle +(1, 1);
                \coordinate ({input cell 5}) at (2.5, 3.5);

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (2, 4) rectangle +(1, 1);
                \coordinate ({input cell 6}) at (2.5, 4.5);

                \draw[input cell,
                    fill opacity={min(rnd + 0.2, 1.0)}]
                    (3, 4) rectangle +(1, 1);
                \coordinate ({input cell 7}) at (3.5, 4.5);

            \end{scope}
        \end{pgfonlayer}

        % Input flow directions grid -----------------------------------------------
        \begin{pgfonlayer}{foreground}
            \begin{scope}[
                    yshift=(3 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[input sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                \foreach \p [count=\i] in \mypoints {

                    % \draw[input cell, fill=none] \p rectangle +(1, 1);

                    % Add 0.5 to the coordinate to arrive at the cell center
                    \tikzmath{
                        coordinate \q;
                        \q = \p + (0.5, 0.5);
                    }

                    \coordinate[at=(\q)] ({input flow direction \i});
                }

                \draw[direction] (0.5, 0.5) -- (1.5, 1.5);
                \draw[direction] (1.5, 0.5) -- (2.5, 1.5);
                \draw[direction] (3.5, 0.5) -- (3.5, 1.5);
                \draw[direction] (4.5, 0.5) -- (4.5, 1.5);

                \draw[direction] (0.5, 1.5) -- (1.5, 1.5);
                \draw[direction] (1.5, 1.5) -- (2.5, 1.5);
                \draw[direction] (2.5, 1.5) -- (3.5, 2.5);
                \draw[direction] (3.5, 1.5) -- (3.5, 2.5);
                \draw[direction] (4.5, 1.5) -- (4.5, 2.5);

                \draw[direction] (0.5, 2.5) -- (0.5, 1.5);

                \draw[direction] (3.5, 2.5) -- (4.5, 2.5);

                \draw[direction] (1.5, 3.5) -- (2.5, 2.5);
                \draw[direction] (2.5, 3.5) -- (2.5, 2.5);
                \draw[direction] (3.5, 3.5) -- (3.5, 2.5);
                \draw[direction] (4.5, 3.5) -- (4.5, 2.5);

                \draw[direction] (0.5, 4.5) -- (1.5, 3.5);
                \draw[direction] (1.5, 4.5) -- (1.5, 3.5);
                \draw[direction] (2.5, 4.5) -- (1.5, 3.5);
                \draw[direction] (3.5, 4.5) -- (2.5, 3.5);
                \draw[direction] (4.5, 4.5) -- (4.5, 3.5);

            \end{scope}
        \end{pgfonlayer}

        \begin{pgfonlayer}{main}
            \foreach \p [count=\i] in \mypoints {
                \draw[arrow]
                    (input flow direction \i) -- (input cell \i);
            }
        \end{pgfonlayer}

        \begin{pgfonlayer}{background}
            \foreach \p [count=\i] in \mypoints {
                \draw[arrow, shorten >= \myscale * 1.0cm]
                    (input cell \i) -- (output cell);
            }
        \end{pgfonlayer}

    \end{tikzpicture}
