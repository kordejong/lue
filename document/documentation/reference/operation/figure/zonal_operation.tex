\documentclass[background=light]{nord_figure}

\begin{document}
    \input{operation.tex}


    \begin{tikzpicture}[
            >=stealth,
            scale=\myscale,
        ]

        \def\mypoints{(1, 1), (1, 2), (2, 2), (3, 1), (3, 2), (2, 4)}

        % Output grid --------------------------------------------------------------
        \begin{pgfonlayer}{background}
            \begin{scope}[
                    yshift=(0 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[output sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                \draw[output cell] (2, 2) rectangle +(1, 1);

                \coordinate (output cell) at (2.5, 2.5);

            \end{scope}
        \end{pgfonlayer}

        % Input values grid --------------------------------------------------------
        \begin{pgfonlayer}{main}
            \begin{scope}[
                    yshift=(2 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[input sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                \foreach \p [count=\i] in \mypoints {
                    \draw[input cell, fill opacity={min(rnd + 0.2, 1.0)}]
                        \p rectangle +(1, 1);

                    % Add 0.5 to the coordinate to arrive at the cell center
                    \tikzmath{
                        coordinate \q;
                        \q = \p + (0.5, 0.5);
                    }

                    \coordinate[at=(\q)] ({input cell \i});
                }

            \end{scope}
        \end{pgfonlayer}

        % Input classes grid -------------------------------------------------------
        \begin{pgfonlayer}{foreground}
            \begin{scope}[
                    yshift=(3 * \mysheetspacing),
                    yslant=\myyslant,
                    xslant=\myxslant,
                ]

                \fill[input sheet] (-0.5, -0.5) rectangle (5.5, 5.5);
                \draw[grid] (-0.25, -0.25) grid (5.25, 5.25);

                \foreach \p [count=\i] in \mypoints {
                    \draw[input cell, fill=nord purple]
                        \p rectangle +(1, 1);

                    % Add 0.5 to the coordinate to arrive at the cell center
                    \tikzmath{
                        coordinate \q;
                        \q = \p + (0.5, 0.5);
                    }

                    \coordinate[at=(\q)] ({input class \i});
                }

                \draw[input cell, fill=nord green] (0, 0) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (0, 1) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (1, 0) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (1, 3) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (2, 0) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (2, 1) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (3, 0) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (4, 0) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (4, 1) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (4, 2) rectangle +(1, 1);
                \draw[input cell, fill=nord green] (4, 3) rectangle +(1, 1);

                \draw[input cell, fill=nord orange] (0, 2) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (0, 3) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (0, 4) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (1, 4) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (2, 3) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (3, 3) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (3, 4) rectangle +(1, 1);
                \draw[input cell, fill=nord orange] (4, 4) rectangle +(1, 1);

            \end{scope}
        \end{pgfonlayer}

        \begin{pgfonlayer}{main}
            \foreach \p [count=\i] in \mypoints {
                \draw[arrow]
                    (input class \i) -- (input cell \i);
            }
        \end{pgfonlayer}

        \begin{pgfonlayer}{background}
            \foreach \p [count=\i] in \mypoints {
                \draw[arrow, shorten >= \myscale * 1.0cm]
                    (input cell \i) -- (output cell);
            }
        \end{pgfonlayer}

    \end{tikzpicture}
\end{document}
